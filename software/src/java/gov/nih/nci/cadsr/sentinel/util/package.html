This is the package for the Sentinel Tool Application Programming Interface (API). Following is example code to use the published features of the API.

// Instantiate the Sentinel Interface
Connection con ... // The caDSR database connection.
String user ... // The caDSR user id which will own the new Alert Definition.
String idseq ... // The database id of the record to watch.
DSRAlert sentinel = DSRAlertImpl.factory(con);

// Ask the Sentinel Tool to create a new Alert Definition.
int rc = sentinel.createAlert(user, idseq);

// Check the results of the request.
switch (rc)
{

case DSRAlert.RC_FAILED:
msg = "An error occurred attempting to create the Alert Definition.";
break;

case DSRAlert.RC_INCOMPATIBLE:
msg = "The Sentinel API server does not support this request.";
break;

case DSRAlert.RC_UNAUTHORIZED:
msg = "You are not authorized to create a Sentinel Alert.";
break;

default:
msg = "The object is monitored by the Alert Definition \"" + sentinel.getAlertName() + "\"";
break;

}
